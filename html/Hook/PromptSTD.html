def clear_screen(msg = nil)
  puts "\e[H\e[2J" if $stdout.tty?
  puts msg if msg.good?
end

def silence_std(file = '/dev/null')
  $stdout = File.new(file, 'w')
  $stderr = File.new(file, 'w')
end
