def clip
  res = `/bin/echo -n #{Shellwords.escape(self)} | pbcopy`.strip

def split_hook
  elements = split(/\|\|/)
  {
    name: elements[0].nil_if_missing,
    url: elements[1].nil_if_missing,
    path: elements[2].nil_if_missing
  }
end

def valid_hook
  if File.exist?(self)
    File.expand_path(self)
  elsif self =~ /^\[.*?\]\((.*?)\)$/
    mdlink = $1
    mdlink.valid_hook
  else
    self
  end
end
